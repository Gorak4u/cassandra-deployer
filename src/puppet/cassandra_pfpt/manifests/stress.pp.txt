# @summary Manages the installation of Cassandra stress testing tools and configuration.
class cassandra_pfpt::stress (
  String $user,
  Sensitive[String] $password,
  Boolean $ssl_enabled,
) {
  # The cassandra-tools package is managed by the cassandra_pfpt::install class.
  # We just need to ensure our resources depend on it.

  # Create a secure configuration file for the stress test wrapper script
  file { '/etc/cassandra/conf/stress.conf':
    ensure  => file,
    owner   => 'root',
    group   => 'root',
    mode    => '0600',
    content => template('cassandra_pfpt/stress.conf.erb'),
    require => Package['cassandra-tools'],
  }

  # Deploy the default stress schema file
  file { '/etc/cassandra/conf/stress-schema.yaml':
    ensure  => file,
    owner   => 'root',
    group   => 'root',
    mode    => '0644',
    source  => 'puppet:///modules/cassandra_pfpt/stress-schema.yaml',
    require => Package['cassandra-tools'],
  }
}
